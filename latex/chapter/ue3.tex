\subsection{b}\label{ass3_t1_b}

\begin{equation}
  v_{B}
  =
  \begin{pmatrix}
    1 \\
    -2\\
    3
  \end{pmatrix}
\end{equation}
\begin{equation}
  ^{A}v_{B}
  =
  ^{B}_{A}T\cdot v_{B}
  =
  \begin{pmatrix}
    0   & -1  & 0   & -3  \\
    -1  & 0   & 0   & -2  \\
    0   & 0   & -1  & -2  \\
    0   & 0   & 0   & 1
  \end{pmatrix}
  \cdot
  \begin{pmatrix}
    1 \\
    -2\\
    3 \\
    1
  \end{pmatrix}
  =
  \begin{pmatrix}
    0\cdot 1  &+& -1\cdot -2 &+& 0\cdot 3  &+& -3\cdot 1 \\
    -1\cdot 1 &+& 0\cdot -2  &+& 0\cdot 3  &+& -2\cdot 1 \\
    0\cdot 1  &+& 0\cdot -2  &+& -1\cdot 3 &+& -2\cdot 1 \\
    0\cdot 1  &+& 0\cdot -2  &+& 0\cdot 3  &+& 1\cdot 1
  \end{pmatrix}
  =
  \begin{pmatrix}
    -1 \\
    -3\\
    -5 \\
    1
  \end{pmatrix}
\end{equation}
